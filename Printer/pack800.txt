    private string oper10to33(int data)
    {
      string str1 = "";
      int num = data;
      string str2;
      if (data < 33)
      {
        str2 = PACK800C._33Index[data];
      }
      else
      {
        int index1;
        while (true)
        {
          index1 = num / 33;
          int index2 = num % 33;
          str1 = PACK800C._33Index[index2] + str1;
          if (index1 >= 33)
            num = index1;
          else
            break;
        }
        str2 = PACK800C._33Index[index1] + str1;
      }
      return str2;
    }

    private void serialPort1_DataReceived(object sender, SerialDataReceivedEventArgs e)
    {
      if (this.serialPort1.IsOpen)
      {
        string Text = this.serialPort1.ReadExisting();
        if (Text != string.Empty)
        {
          Text.ToCharArray();
          this.DataProcessing(Text);
        }
      }
      if (this.txtCONT_SRIAL.Text.Length != 14)
        return;
      this.MainProcess();
      this.DataProcessing("1");
    }
